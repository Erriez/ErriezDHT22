122 int16_t DHT22::readTemperature()
123 {
124  int16_t temperature = ~0;
125 
126  // Read data from sensor
127  if (_statusLastMeasurement != true) {
128  return temperature;
129  }
130 
131  // Calculate signed temperature
132  temperature = ((_data[2] & 0x7F) << 8) | _data[3];
133  if (_data[2] & 0x80) {
134  temperature *= -1;
135  }
136 
137  // Calculate temperature average
138  if ((_temperatureSamples != NULL) && (temperature != ~0)) {
139  // Store temperature sample
140  _temperatureSamples[_temperatureSampleIndex++ % _numSamples] = temperature;
141 
142  // Increment number of samples
143  if (_numTemperatureSamples < _numSamples) {
144  _numTemperatureSamples++;
145  }
146 
147  // Calculate average temperature
148  temperature = 0;
149  for (uint8_t i = 0; i < _numTemperatureSamples; i++) {
150  temperature += _temperatureSamples[i];
151  }
152  temperature /= _numTemperatureSamples;
153  }
154 
155  return temperature;
156 }

214 bool DHT22::readSensorData()
215 {
216  // Store last conversion timestamp
217  _lastMeasurementTimestamp = millis();
218 
219  // Read data from sensor until valid data has been read or maximum number of retries
220  // Mark current measurement as successful
221  _statusLastMeasurement = true;
222 
223  // Generate sensor start pulse
224  if (generateStart() != true) {
225  DEBUG_PRINTLN(F("DHT22: Start error"));
226  // Mark measurement as invalid
227  _statusLastMeasurement = false;
228  }
229 
230  // Read 5 Bytes data from sensor
231  if (_statusLastMeasurement) {
232  if (readBytes() != true) {
233  DEBUG_PRINTLN(F("DHT22: Read error"));
234  // Mark measurement as invalid
235  _statusLastMeasurement = false;
236  }
237  }
238 
239  // Check data parity
240  if (_statusLastMeasurement) {
241  if (((_data[0] + _data[1] + _data[2] + _data[3]) & 0xFF) != _data[4]) {
242  DEBUG_PRINTLN(F("DHT22: Parity error"));
243  // Mark measurement as invalid
244  _statusLastMeasurement = false;
245  }
246  }
247 
248  return _statusLastMeasurement;
249 }

261 bool DHT22::generateStart()
262 {
263  // Data pin high (pull-up)
264  digitalWrite(_pin, HIGH);
265  delay(10);
266 
267  // Change data pin to output, low, followed by high
268  pinMode(_pin, OUTPUT);
269  digitalWrite(_pin, LOW);
270  delay(20);
271 
272  // Data pin to input (pull-up)
273  pinMode(_pin, INPUT_PULLUP);
274  delayMicroseconds(30);
275 
276  // Check data pin timing low
277  if (measurePulseWidth(LOW) == 0) {
278  return false;
279  }
280 
281  // Check data pin timing high
282  if (measurePulseWidth(HIGH) == 0) {
283  return false;
284  }
285 
286  // Sensor start successfully generated
287  return true;
288 }

300 bool DHT22::readBytes()
301 {
302  // Disable interrupts during data transfer
303  noInterrupts();
304 
305  // Measure and store pulse width of each bit
306  for (int i = 0; i < (DHT22_NUM_DATA_BITS * 2); i += 2) {
307  cycles[i] = measurePulseWidth(LOW);
308  cycles[i + 1] = measurePulseWidth(HIGH);
309  }
310 
311  // Enable interrupts
312  interrupts();
313 
314  // Clear data buffer
315  memset(_data, 0, sizeof(_data));
316 
317  // Convert pulse width to data bit
318  for (int i = 0; i < DHT22_NUM_DATA_BITS; ++i) {
319  uint32_t lowCycles = cycles[2 * i];
320  uint32_t highCycles = cycles[(2 * i) + 1];
321 
322  // Check valid bit timing
323  if ((lowCycles == 0) || (highCycles == 0)) {
324  return false;
325  }
326 
327  // Calculate byte index in data array
328  int byteIndex = i / 8;
329 
330  // Store bit
331  _data[byteIndex] <<= 1;
332  if (highCycles > lowCycles) {
333  _data[byteIndex] |= 1;
334  }
335  }
336 
337  return true;
338 }

348 uint32_t DHT22::measurePulseWidth(uint8_t level)
349 {
350  uint32_t count = 0;
351 
352 #ifdef __AVR
353  while ((*portInputRegister(_port) & _bit) == (level ? _bit : 0)) {
354  if (count++ >= _maxCycles) {
355  // Timeout
356  return 0;
357  }
358  }
359 
360 #else
361  while (digitalRead(_pin) == level) {
362  if (count++ >= _maxCycles) {
363  // Timeout
364  return 0;
365  }
366  }
367 #endif
368 
369  return count;
