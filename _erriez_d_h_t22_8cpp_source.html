182 bool DHT22::readSensorData()
183 {
184  if (available() == false) {
185  // Return last measurement status
186  return _statusLastMeasurement;
187  }
188 
189  // Store last conversion timestamp
190  _lastMeasurementTimestamp = millis();
191 
192  // Read data from sensor until valid data has been read or maximum number of retries
193  for (_numReadRetries = 0; _numReadRetries <= _maxReadRetries; _numReadRetries++) {
194  // Mark current measurement as successful
195  _statusLastMeasurement = true;
196 
197  // Generate sensor start pulse
198  if (generateStart() != true) {
199  DEBUG_PRINTLN(F("DHT22: Start error"));
200  // Mark measurement as invalid
201  _statusLastMeasurement = false;
202  }
203 
204  // Read 5 Bytes data from sensor
205  if (_statusLastMeasurement) {
206  if (readBytes() != true) {
207  DEBUG_PRINTLN(F("DHT22: Read error"));
208  // Mark measurement as invalid
209  _statusLastMeasurement = false;
210  }
211  }
212 
213  // Check data parity
214  if (_statusLastMeasurement) {
215  if (((_data[0] + _data[1] + _data[2] + _data[3]) & 0xFF) != _data[4]) {
216  DEBUG_PRINTLN(F("DHT22: Parity error"));
217  // Mark measurement as invalid
218  _statusLastMeasurement = false;
219  }
220  }
221 
222  // Check valid conversion
223  if (_statusLastMeasurement == true) {
224  break;
225  }
226  }
227 
228  // Fix number of retries when exceeding max reads in for loop
229  _numReadRetries = min(_numReadRetries, _maxReadRetries);
230 
231  return _statusLastMeasurement;
232 }

241 bool DHT22::generateStart()
242 {
243  // Data pin high (pull-up)
244  digitalWrite(_pin, HIGH);
245  delay(10);
246 
247  // Change data pin to output, low, followed by high

280 bool DHT22::readBytes()
281 {
282  // Disable interrupts during data transfer
283  noInterrupts();
284 
285  // Measure and store pulse width of each bit
286  for (int i = 0; i < (DHT22_NUM_DATA_BITS * 2); i += 2) {
287  cycles[i] = measurePulseWidth(LOW);
288  cycles[i + 1] = measurePulseWidth(HIGH);
289  }
290 
291  // Enable interrupts
292  interrupts();
293 
294  // Clear data buffer
295  memset(_data, 0, sizeof(_data));
296 
297  // Convert pulse width to data bit
298  for (int i = 0; i < DHT22_NUM_DATA_BITS; ++i) {
299  uint32_t lowCycles = cycles[2 * i];
300  uint32_t highCycles = cycles[(2 * i) + 1];
301 
302  // Check valid bit timing
303  if ((lowCycles == 0) || (highCycles == 0)) {
304  return false;
305  }
306 
307  // Calculate byte index in data array
308  int byteIndex = i / 8;
309 
310  // Store bit
311  _data[byteIndex] <<= 1;
312  if (highCycles > lowCycles) {
313  _data[byteIndex] |= 1;
314  }
315  }
316 
317  return true;
318 }
